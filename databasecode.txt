CREATE TABLE chatbot (
	id int not null AUTO_INCREMENT,
    replies text DEFAULT NULL,
    queries text DEFAULT NULL,
    PRIMARY KEY (id)
);

	id int not null AUTO_INCREMENT,
    fromUser INT DEFAULT NULL,
    toUser INT DEFAULT NULL,
    message text,
    PRIMARY KEY (id)
);
